	} else {
		// 조건식이 거짓 일때 실행되는 부분
	}
	
	if(조건식1){
	
	} else if(조건식2) {
		// 조건식 1이거짓, 조건식2 참
	} else if(조건식3){
		// 조건식 1거짓, 조건식2 거짓, 조건식3 참
	}
	
	조건식은 참, 거짓(논리값)
	비교 연산자 + 논리 연산자
	(크기, 작다, 같다)			(||-OR, && - AND, ! - NOT)
