		if(조건식1) {
				//	조건식이 참 일때 실행

				//	조건식1이 거짓이고 조건식2가 참일때 실행 영역

				//	조건식1이 거짓이고 조건식2가 거짓, 조건식3이 참 일때 실행되는 영역
		}		else {
				//	모든 조건식이 거짓일 때 실행되는 영역
		}
		
		조건식을 여러개 비교 --> 논리 연산자와 결합
		if(조건식1 || 조건식2) {
			// 조건식1 또는 조건식2가 참이면

	- 반복문
		반복을 할때 쓰는 구문
		
		while문
		while(조건식) {
			// 조건식이 참일때 반복이 되는 영역
		}
		
		break -- 반복 멈춤
		
		continue -- 반복 건너뜀 (현재의 반복을 중단하고 다시 새로 반복)
		짝수
			0, 2, 4, 6, 8, 10 .... 2로 나누어지는 수
				-> 2로 나누었을때 나머지가 0
				a % 2== 0  -> 짝수
		
		var a=0;
			while(a<100) {
				a++
			if(a%2==1){
				continue;  // 홀수 이면 건너 뜀
				}
				console.log(a)
			}
			
		do~while문
		
		do {
			// 조건식이 참일때 반복이 되는 영역
			
		} while(조건식);
	
			-> 우선 먼저 시도 하고 조건식에 따라 반복
			-> 최소한 1번은 실핼 됨
			-> 최소한 1번은 무조건 실행이 되어야 하는 경우
		
		
		for문
			
			for(초기값; 조건식; 증감식){
				//  조건식이 true 일 때 까지 반복
			}
